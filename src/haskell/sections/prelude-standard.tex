%%% Numeric type classes
\textbf{Num} is a basic numeric class.
The Haskell Report defines no laws for Num.
However, (+) and (*) are customarily expected to define a ring.
\begin{minted}{haskell}
class  Num a  where
    (+), (-), (*)       :: a -> a -> a
    negate              :: a -> a
    abs                 :: a -> a
    signum              :: a -> a
    fromInteger         :: Integer -> a
    x - y               = x + negate y
    negate x            = 0 - x
\end{minted}
