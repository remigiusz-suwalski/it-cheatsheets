\subsection{requests -- simple yet elegant HTTP library (pypi)}
\begin{minted}{python}
import requests
r = requests.get('https://api.github.com/user',
    auth=('user', 'pass'))
r.status_code
# 200
r.headers['content-type']
# 'application/json; charset=utf8'
r.encoding
# 'utf-8'
r.text
# '{"type":"User"...'
r.json()
# {'disk_usage': 368627, 'private_gists': 484, ...}
\end{minted}
